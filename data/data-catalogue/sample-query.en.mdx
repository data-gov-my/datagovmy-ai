```swift
import Foundation
#if canImport(FoundationNetworking)
import FoundationNetworking
#endif

var request = URLRequest(url: URL(string: "https://api.data.gov.my/data-catalogue?id=fuelprice&limit=3")!,timeoutInterval: Double.infinity)
request.httpMethod = "GET"

let task = URLSession.shared.dataTask(with: request) { data, response, error in
  guard let data = data else {
    print(String(describing: error))
    exit(EXIT_SUCCESS)
  }
  print(String(data: data, encoding: .utf8)!)
  exit(EXIT_SUCCESS)
}

task.resume()
```

### Querying in Dart

The following code is an example of how to make an API query in Dart to retrieve data from the data catalogue endpoint. For a complete guide on possible query parameters and syntax, please refer to the [Request Query documentation](https://developer.data.gov.my/data-catalogue/request-query). Additionally, you can experiment with the code interactively and send requests at the [Example Requests section](https://developer.data.gov.my/data-catalogue/example-requests).

```dart
import 'package:http/http.dart' as http;

void main() async {
  var request = http.Request('GET', Uri.parse('https://api.data.gov.my/data-catalogue?id=fuelprice&limit=3'));

  request.followRedirects = false;

  http.StreamedResponse response = await request.send();

  if (response.statusCode == 200) {
    print(await response.stream.bytesToString());
  }
  else {
    print(response.reasonPhrase);
  }

}
```

### Querying in Java

The following code is an example of how to make an API query in Java to retrieve data from the data catalogue endpoint. For a complete guide on possible query parameters and syntax, please refer to the [Request Query documentation](https://developer.data.gov.my/data-catalogue/request-query). Additionally, you can experiment with the code interactively and send requests at the [Example Requests section](https://developer.data.gov.my/data-catalogue/example-requests).

```java
import java.io.*;
import okhttp3.*;
public class Main {
  public static void main(String []args) throws IOException{
    OkHttpClient client = new OkHttpClient().newBuilder()
      .followRedirects(false)
      .build();
    MediaType mediaType = MediaType.parse("text/plain");
    RequestBody body = RequestBody.create(mediaType, "");
    Request request = new Request.Builder()
      .url("https://api.data.gov.my/data-catalogue?id=fuelprice&limit=3")
      .method("GET", body)
      .build();
    Response response = client.newCall(request).execute();
    System.out.println(response.body().string());
  }
}
```
